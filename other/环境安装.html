# 启动、停止、重启服务

# 查看版本
sudo docker version
# 加入开机自启
sudo systemctl enable | disable docker

# 升级与卸载

# 升级
# sudo apt-get update

# 卸载

# 删除临时文件
sudo rm -rf /var/lib/docker
# 删除依赖
sudo apt autoremove
